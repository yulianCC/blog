def rc4_init(key):
    s = list(range(256))
    j = 0
    key_len = len(key)
    
    for i in range(256):
        j = (j + s[i] + key[i % key_len]) % 256
        s[i], s[j] = s[j], s[i] 
    return s

def rc4_crypt1(s, data):
    i, j = 0, 0
    res = []
    for char in data:
        i = (i + 1) % 256
        j = (j + s[i]) % 256
        s[i], s[j] = s[j], s[i]
        k = s[(s[i] + s[j]) % 256]
        res.append(char ^ k)
    return res

def rc4_crypt2(s, data):
    i, j = 0, 0
    res = []
    for char in data:
        i = (i + 1) % 256
        j = (j + s[i]) % 256
        s[i], s[j] = s[j], s[i]
        k = s[(s[i] + s[j]) % 256]
        res.append((char + k) % 256) 
    return res

key1 = [ord(b) for b in "keykey"]
